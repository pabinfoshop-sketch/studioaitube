   132‚Üí   132‚Üífunction renderGlitch(ctx: CanvasRenderingContext2D, W: number, H: number, t: number, frame: number) {
   133‚Üí   133‚Üí  // Random glitch bars
   134‚Üí   134‚Üí  const glitchIntensity = Math.random() > 0.85 ? 0.8 : 0.15;
   135‚Üí   135‚Üí  const numBars = Math.floor(Math.random() * 8 * glitchIntensity) + 1;
   136‚Üí   136‚Üí  for (let i = 0; i < numBars; i++) {
   137‚Üí   137‚Üí    const y = Math.random() * H;
   138‚Üí   138‚Üí    const barH = Math.random() * 20 + 2;
   139‚Üí   139‚Üí    const offset = (Math.random() - 0.5) * 40 * glitchIntensity;
   140‚Üí   140‚Üí    try {
   141‚Üí   141‚Üí      const imgData = ctx.getImageData(0, Math.max(0, Math.floor(y)), W, Math.min(Math.ceil(barH), Math.floor(H - y)));
   142‚Üí   142‚Üí      ctx.putImageData(imgData, offset, Math.floor(y));
   143‚Üí   143‚Üí    } catch { /* cross-origin safety */ }
   144‚Üí   144‚Üí  }
   145‚Üí   145‚Üí  // RGB shift
   146‚Üí   146‚Üí  if (Math.random() > 0.7) {
   147‚Üí   147‚Üí    ctx.globalCompositeOperation = "screen";
   148‚Üí   148‚Üí    ctx.fillStyle = `rgba(255, 0, 0, ${0.03 * glitchIntensity})`;

   150‚Üí   150‚Üí    ctx.fillStyle = `rgba(0, 255, 255, ${0.03 * glitchIntensity})`;

   306‚Üí   306‚Üíexport function renderCaption(
   307‚Üí   307‚Üí  ctx: CanvasRenderingContext2D,
   308‚Üí   308‚Üí  W: number,
   309‚Üí   309‚Üí  H: number,
   310‚Üí   310‚Üí  currentTime: number, // seconds into the scene
   311‚Üí   311‚Üí  segments: CaptionSegment[],
   312‚Üí   312‚Üí  style: CaptionStyle,
   313‚Üí   313‚Üí  frame: number,
   314‚Üí   314‚Üí) {
   315‚Üí   315‚Üí  if (style === "none" || !segments.length) return;
   316‚Üí   316‚Üí
   317‚Üí   317‚Üí  // Find active segment
   318‚Üí   318‚Üí  let activeSeg: CaptionSegment | null = null;
   319‚Üí   319‚Üí  let segProgress = 0;
   320‚Üí   320‚Üí  for (const seg of segments) {
   321‚Üí   321‚Üí    if (currentTime >= seg.start && currentTime <= seg.end) {
   322‚Üí   322‚Üí      activeSeg = seg;
   323‚Üí   323‚Üí      segProgress = (currentTime - seg.start) / (seg.end - seg.start || 1);
   324‚Üí   324‚Üí      break;
   325‚Üí   325‚Üí    }
   326‚Üí   326‚Üí  }
   327‚Üí   327‚Üí  if (!activeSeg) return;
   328‚Üí   328‚Üí
   329‚Üí   329‚Üí  const text = activeSeg.text.trim();
   330‚Üí   330‚Üí  if (!text) return;
   331‚Üí   331‚Üí
   332‚Üí   332‚Üí  switch (style) {
   333‚Üí   333‚Üí    case "tiktok":
   334‚Üí   334‚Üí      renderTikTokCaption(ctx, W, H, text);
   335‚Üí   335‚Üí      break;
   336‚Üí   336‚Üí    case "cinematic":
   337‚Üí   337‚Üí      renderCinematicCaption(ctx, W, H, text);
   338‚Üí   338‚Üí      break;
   339‚Üí   339‚Üí    case "karaoke":
   340‚Üí   340‚Üí      renderKaraokeCaption(ctx, W, H, text, segProgress, activeSeg, currentTime);
   341‚Üí   341‚Üí      break;
   342‚Üí   342‚Üí    case "glitch-text":
   343‚Üí   343‚Üí      renderGlitchCaption(ctx, W, H, text, frame);
   344‚Üí   344‚Üí      break;
   345‚Üí   345‚Üí  }
   346‚Üí   346‚Üí}

   413‚Üí   413‚Üífunction renderKaraokeCaption(
   414‚Üí   414‚Üí  ctx: CanvasRenderingContext2D,
   415‚Üí   415‚Üí  W: number,
   416‚Üí   416‚Üí  H: number,

   418‚Üí   418‚Üí  progress: number,

   442‚Üí   442‚Üí  const fullWidth = ctx.measureText(fullText).width;
   443‚Üí   443‚Üí  const spokenWidth = ctx.measureText(spokenText).width;
   444‚Üí   444‚Üí  const startX = W / 2 - fullWidth / 2;

   448‚Üí   448‚Üí  ctx.beginPath();
   449‚Üí   449‚Üí  ctx.rect(startX, y - fontSize, spokenWidth, fontSize * 2);

   451‚Üí   451‚Üí  ctx.fillStyle = "#a78bfa";
   452‚Üí   452‚Üí  ctx.fillText(fullText, W / 2, y);
   453‚Üí   453‚Üí  ctx.restore();
   454‚Üí   454‚Üí
   455‚Üí   455‚Üí  // Background bar
   456‚Üí   456‚Üí  ctx.fillStyle = "rgba(0,0,0,0.6)";
   457‚Üí   457‚Üí  const barW = fullWidth + 40;
   458‚Üí   458‚Üí  roundRect(ctx, W / 2 - barW / 2, y - fontSize / 2 - 8, barW, fontSize + 16, 10);
   459‚Üí   459‚Üí  ctx.fill();
   460‚Üí   460‚Üí
   461‚Üí   461‚Üí  // Re-draw text on top of background
   462‚Üí   462‚Üí  ctx.fillStyle = "rgba(255,255,255,0.4)";
   463‚Üí   463‚Üí  ctx.fillText(fullText, W / 2, y);
   464‚Üí   464‚Üí  ctx.save();
   465‚Üí   465‚Üí  ctx.beginPath();
   466‚Üí   466‚Üí  ctx.rect(startX, y - fontSize, spokenWidth, fontSize * 2);
   467‚Üí   467‚Üí  ctx.clip();
   468‚Üí   468‚Üí  ctx.fillStyle = "#FFFFFF";
   469‚Üí   469‚Üí  ctx.fillText(fullText, W / 2, y);
   470‚Üí   470‚Üí  ctx.restore();
   471‚Üí   471‚Üí}

   501‚Üí   501‚Üí// ‚îÄ‚îÄ Audio Effect Processors (Web Audio API) ‚îÄ‚îÄ
   502‚Üí   502‚Üí
   503‚Üí   503‚Üíexport function applyVoiceEffect(
   504‚Üí   504‚Üí  audioCtx: AudioContext,
   505‚Üí   505‚Üí  sourceNode: AudioNode,
   506‚Üí   506‚Üí  effect: VoiceEffect,
   507‚Üí   507‚Üí): { chain: AudioNode; destination: AudioNode } {
   508‚Üí   508‚Üí  if (effect === "none") {
   509‚Üí   509‚Üí    return { chain: sourceNode, destination: audioCtx.destination };
   510‚Üí   510‚Üí  }
   511‚Üí   511‚Üí
   512‚Üí   512‚Üí  const dest = audioCtx.createMediaStreamDestination();
   513‚Üí   513‚Üí  const gainNode = audioCtx.createGain();
   514‚Üí   514‚Üí  gainNode.gain.value = 1.0;
   515‚Üí   515‚Üí
   516‚Üí   516‚Üí  switch (effect) {
   517‚Üí   517‚Üí    case "echo": {
   518‚Üí   518‚Üí      const delay = audioCtx.createDelay(1.0);
   519‚Üí   519‚Üí      delay.delayTime.value = 0.3;
   520‚Üí   520‚Üí      const feedback = audioCtx.createGain();
   521‚Üí   521‚Üí      feedback.gain.value = 0.5;
   522‚Üí   522‚Üí      const wetGain = audioCtx.createGain();
   523‚Üí   523‚Üí      wetGain.gain.value = 0.4;
   524‚Üí   524‚Üí
   525‚Üí   525‚Üí      sourceNode.connect(gainNode);
   526‚Üí   526‚Üí      gainNode.connect(dest);
   527‚Üí   527‚Üí      gainNode.connect(delay);
   528‚Üí   528‚Üí      delay.connect(feedback);
   529‚Üí   529‚Üí      feedback.connect(delay);
   530‚Üí   530‚Üí      delay.connect(wetGain);
   531‚Üí   531‚Üí      wetGain.connect(dest);
   532‚Üí   532‚Üí      gainNode.connect(audioCtx.destination);
   533‚Üí   533‚Üí      return { chain: gainNode, destination: dest };
   534‚Üí   534‚Üí    }
   535‚Üí   535‚Üí    case "reverb": {
   536‚Üí   536‚Üí      // Create convolution reverb (synthetic impulse response)
   537‚Üí   537‚Üí      const convolver = audioCtx.createConvolver();
   538‚Üí   538‚Üí      const sampleRate = audioCtx.sampleRate;
   539‚Üí   539‚Üí      const length = sampleRate * 2;
   540‚Üí   540‚Üí      const impulse = audioCtx.createBuffer(2, length, sampleRate);
   541‚Üí   541‚Üí      for (let ch = 0; ch < 2; ch++) {
   542‚Üí   542‚Üí        const data = impulse.getChannelData(ch);
   543‚Üí   543‚Üí        for (let i = 0; i < length; i++) {
   544‚Üí   544‚Üí          data[i] = (Math.random() * 2 - 1) * Math.pow(1 - i / length, 2.5);
   545‚Üí   545‚Üí        }
   546‚Üí   546‚Üí      }
   547‚Üí   547‚Üí      convolver.buffer = impulse;
   548‚Üí   548‚Üí
   549‚Üí   549‚Üí      const wetGain = audioCtx.createGain();
   550‚Üí   550‚Üí      wetGain.gain.value = 0.35;
   551‚Üí   551‚Üí
   552‚Üí   552‚Üí      sourceNode.connect(gainNode);
   553‚Üí   553‚Üí      gainNode.connect(dest);
   554‚Üí   554‚Üí      gainNode.connect(convolver);
   555‚Üí   555‚Üí      convolver.connect(wetGain);
   556‚Üí   556‚Üí      wetGain.connect(dest);
   557‚Üí   557‚Üí      gainNode.connect(audioCtx.destination);
   558‚Üí   558‚Üí      return { chain: gainNode, destination: dest };
   559‚Üí   559‚Üí    }
   560‚Üí   560‚Üí    case "deep": {
   561‚Üí   561‚Üí      // Pitch down by playing slower ‚Äî approximate with a simple lowpass
   562‚Üí   562‚Üí      const filter = audioCtx.createBiquadFilter();
   563‚Üí   563‚Üí      filter.type = "lowpass";
   564‚Üí   564‚Üí      filter.frequency.value = 2000;
   565‚Üí   565‚Üí      filter.Q.value = 1;
   566‚Üí   566‚Üí
   567‚Üí   567‚Üí      sourceNode.connect(gainNode);
   568‚Üí   568‚Üí      gainNode.connect(filter);
   569‚Üí   569‚Üí      filter.connect(dest);
   570‚Üí   570‚Üí      gainNode.connect(audioCtx.destination);
   571‚Üí   571‚Üí      return { chain: gainNode, destination: dest };
   572‚Üí   572‚Üí    }
   573‚Üí   573‚Üí    case "whisper-voice": {
   574‚Üí   574‚Üí      // Bandpass for whispery effect
   575‚Üí   575‚Üí      const bp = audioCtx.createBiquadFilter();
   576‚Üí   576‚Üí      bp.type = "bandpass";
   577‚Üí   577‚Üí      bp.frequency.value = 3000;
   578‚Üí   578‚Üí      bp.Q.value = 0.5;
   579‚Üí   579‚Üí
   580‚Üí   580‚Üí      const noiseGain = audioCtx.createGain();
   581‚Üí   581‚Üí      noiseGain.gain.value = 0.1;
   582‚Üí   582‚Üí
   583‚Üí   583‚Üí      sourceNode.connect(gainNode);
   584‚Üí   584‚Üí      gainNode.connect(bp);
   585‚Üí   585‚Üí      bp.connect(noiseGain);
   586‚Üí   586‚Üí      noiseGain.connect(dest);
   587‚Üí   587‚Üí      gainNode.connect(audioCtx.destination);
   588‚Üí   588‚Üí      return { chain: gainNode, destination: dest };
   589‚Üí   589‚Üí    }
   590‚Üí   590‚Üí    case "radio": {
   591‚Üí   591‚Üí      // Bandpass + distortion
   592‚Üí   592‚Üí      const bp = audioCtx.createBiquadFilter();
   593‚Üí   593‚Üí      bp.type = "bandpass";
   594‚Üí   594‚Üí      bp.frequency.value = 2500;
   595‚Üí   595‚Üí      bp.Q.value = 3;
   596‚Üí   596‚Üí
   597‚Üí   597‚Üí      const distortion = audioCtx.createWaveShaper();
   598‚Üí   598‚Üí      const curve = new Float32Array(256);
   599‚Üí   599‚Üí      for (let i = 0; i < 256; i++) {
   600‚Üí   600‚Üí        const x = (i * 2) / 256 - 1;
   601‚Üí   601‚Üí        curve[i] = (Math.PI + 3) * x / (Math.PI + 3 * Math.abs(x));
   602‚Üí   602‚Üí      }
   603‚Üí   603‚Üí      distortion.curve = curve;
   604‚Üí   604‚Üí
   605‚Üí   605‚Üí      sourceNode.connect(gainNode);
   606‚Üí   606‚Üí      gainNode.connect(bp);
   607‚Üí   607‚Üí      bp.connect(distortion);
   608‚Üí   608‚Üí      distortion.connect(dest);
   609‚Üí   609‚Üí      gainNode.connect(audioCtx.destination);
   610‚Üí   610‚Üí      return { chain: gainNode, destination: dest };
   611‚Üí   611‚Üí    }
   612‚Üí   612‚Üí    default:
   613‚Üí   613‚Üí      sourceNode.connect(dest);
   614‚Üí   614‚Üí      sourceNode.connect(audioCtx.destination);
   615‚Üí   615‚Üí      return { chain: sourceNode, destination: dest };
   616‚Üí   616‚Üí  }

   626‚Üí   626‚Üí/** Render transition overlay. Returns true if transition is still active. */
   627‚Üí   627‚Üíexport function renderTransition(
   628‚Üí   628‚Üí  ctx: CanvasRenderingContext2D,
   629‚Üí   629‚Üí  W: number,
   630‚Üí   630‚Üí  H: number,
   631‚Üí   631‚Üí  t: number, // 0..1 progress within transition
   632‚Üí   632‚Üí  transition: TransitionEffect,
   633‚Üí   633‚Üí  nextImg?: HTMLImageElement,
   634‚Üí   634‚Üí  nextStyleIndex?: number,
   635‚Üí   635‚Üí): boolean {
   636‚Üí   636‚Üí  if (transition === "none" || t >= 1) return false;
   637‚Üí   637‚Üí
   638‚Üí   638‚Üí  switch (transition) {
   639‚Üí   639‚Üí    case "fade":
   640‚Üí   640‚Üí      ctx.fillStyle = `rgba(0,0,0,${t})`;
   641‚Üí   641‚Üí      ctx.fillRect(0, 0, W, H);
   642‚Üí   642‚Üí      if (t > 0.5 && nextImg) {
   643‚Üí   643‚Üí        ctx.globalAlpha = (t - 0.5) * 2;
   644‚Üí   644‚Üí        ctx.drawImage(nextImg, 0, 0, W, H);
   645‚Üí   645‚Üí        ctx.globalAlpha = 1;
   646‚Üí   646‚Üí      }
   647‚Üí   647‚Üí      return true;
   648‚Üí   648‚Üí
   649‚Üí   649‚Üí    case "glitch-cut": {
   650‚Üí   650‚Üí      // Rapid glitch frames
   651‚Üí   651‚Üí      const numSlices = Math.floor(10 + t * 20);
   652‚Üí   652‚Üí      for (let i = 0; i < numSlices; i++) {
   653‚Üí   653‚Üí        const y = Math.random() * H;
   654‚Üí   654‚Üí        const h = Math.random() * 30 + 5;
   655‚Üí   655‚Üí        ctx.fillStyle = `rgba(${Math.random() > 0.5 ? "139,92,246" : "255,255,255"}, ${0.1 + Math.random() * 0.2})`;
   656‚Üí   656‚Üí        ctx.fillRect((Math.random() - 0.5) * 20, y, W, h);
   657‚Üí   657‚Üí      }
   658‚Üí   658‚Üí      return true;
   659‚Üí   659‚Üí    }
   660‚Üí   660‚Üí
   661‚Üí   661‚Üí    case "flash": {
   662‚Üí   662‚Üí      // White flash that fades
   663‚Üí   663‚Üí      const flashIntensity = t < 0.3 ? t / 0.3 : 1 - ((t - 0.3) / 0.7);
   664‚Üí   664‚Üí      ctx.fillStyle = `rgba(255,255,255,${flashIntensity * 0.9})`;
   665‚Üí   665‚Üí      ctx.fillRect(0, 0, W, H);
   666‚Üí   666‚Üí      return true;
   667‚Üí   667‚Üí    }
   668‚Üí   668‚Üí
   669‚Üí   669‚Üí    case "distortion": {
   670‚Üí   670‚Üí      // Wave distortion effect
   671‚Üí   671‚Üí      try {
   672‚Üí   672‚Üí        const imgData = ctx.getImageData(0, 0, W, H);
   673‚Üí   673‚Üí        const copy = new ImageData(new Uint8ClampedArray(imgData.data), W, H);
   674‚Üí   674‚Üí        const intensity = Math.sin(t * Math.PI) * 15;
   675‚Üí   675‚Üí        for (let y = 0; y < H; y++) {
   676‚Üí   676‚Üí          const offsetX = Math.sin(y * 0.02 + t * 20) * intensity;
   677‚Üí   677‚Üí          const srcY = Math.min(H - 1, Math.max(0, y));
   678‚Üí   678‚Üí          const dstY = Math.min(H - 1, Math.max(0, y));
   679‚Üí   679‚Üí          for (let x = 0; x < W; x++) {
   680‚Üí   680‚Üí            const srcX = Math.min(W - 1, Math.max(0, Math.floor(x + offsetX)));
   681‚Üí   681‚Üí            const dstIdx = (dstY * W + x) * 4;
   682‚Üí   682‚Üí            const srcIdx = (srcY * W + srcX) * 4;
   683‚Üí   683‚Üí            copy.data[dstIdx] = imgData.data[srcIdx];
   684‚Üí   684‚Üí            copy.data[dstIdx + 1] = imgData.data[srcIdx + 1];
   685‚Üí   685‚Üí            copy.data[dstIdx + 2] = imgData.data[srcIdx + 2];
   686‚Üí   686‚Üí          }
   687‚Üí   687‚Üí        }
   688‚Üí   688‚Üí        ctx.putImageData(copy, 0, 0);
   689‚Üí   689‚Üí      } catch { /* cross-origin */ }
   690‚Üí   690‚Üí      return true;
   691‚Üí   691‚Üí    }
   692‚Üí   692‚Üí
   693‚Üí   693‚Üí    case "whisper": {
   694‚Üí   694‚Üí      // Blur-like fade: darken + lighten
   695‚Üí   695‚Üí      const darken = t < 0.5 ? t * 2 : 2 - t * 2;
   696‚Üí   696‚Üí      ctx.fillStyle = `rgba(0,0,0,${darken * 0.7})`;
   697‚Üí   697‚Üí      ctx.fillRect(0, 0, W, H);
   698‚Üí   698‚Üí      if (t > 0.4 && nextImg) {
   699‚Üí   699‚Üí        const alpha = (t - 0.4) / 0.6;
   700‚Üí   700‚Üí        ctx.globalAlpha = alpha * 0.5;
   701‚Üí   701‚Üí        ctx.drawImage(nextImg, 0, 0, W, H);
   702‚Üí   702‚Üí        ctx.globalAlpha = 1;
   703‚Üí   703‚Üí      }
   704‚Üí   704‚Üí      return true;
   705‚Üí   705‚Üí    }
   706‚Üí   706‚Üí
   707‚Üí   707‚Üí    default:
   708‚Üí   708‚Üí      return false;
   709‚Üí   709‚Üí  }
   710‚Üí   710‚Üí}

   714‚Üí   714‚Üífunction wrapText(ctx: CanvasRenderingContext2D, text: string, maxWidth: number): string[] {
   715‚Üí   715‚Üí  const words = text.split(" ");
   716‚Üí   716‚Üí  const lines: string[] = [];
   717‚Üí   717‚Üí  let current = "";
   718‚Üí   718‚Üí  for (const word of words) {
   719‚Üí   719‚Üí    const test = current ? `${current} ${word}` : word;
   720‚Üí   720‚Üí    if (ctx.measureText(test).width > maxWidth && current) {
   721‚Üí   721‚Üí      lines.push(current);
   722‚Üí   722‚Üí      current = word;
   723‚Üí   723‚Üí    } else {
   724‚Üí   724‚Üí      current = test;
   725‚Üí   725‚Üí    }
   726‚Üí   726‚Üí  }
   727‚Üí   727‚Üí  if (current) lines.push(current);
   728‚Üí   728‚Üí  return lines.slice(0, 3); // Max 3 lines
   729‚Üí   729‚Üí}

   731‚Üí   731‚Üífunction roundRect(ctx: CanvasRenderingContext2D, x: number, y: number, w: number, h: number, r: number) {
   732‚Üí   732‚Üí  ctx.beginPath();
   733‚Üí   733‚Üí  ctx.moveTo(x + r, y);
   734‚Üí   734‚Üí  ctx.lineTo(x + w - r, y);
   735‚Üí   735‚Üí  ctx.quadraticCurveTo(x + w, y, x + w, y + r);
   736‚Üí   736‚Üí  ctx.lineTo(x + w, y + h - r);
   737‚Üí   737‚Üí  ctx.quadraticCurveTo(x + w, y + h, x + w - r, y + h);
   738‚Üí   738‚Üí  ctx.lineTo(x + r, y + h);
   739‚Üí   739‚Üí  ctx.quadraticCurveTo(x, y + h, x, y + h - r);
   740‚Üí   740‚Üí  ctx.lineTo(x, y + r);
   741‚Üí   741‚Üí  ctx.quadraticCurveTo(x, y, x + r, y);
   742‚Üí   742‚Üí  ctx.closePath();
