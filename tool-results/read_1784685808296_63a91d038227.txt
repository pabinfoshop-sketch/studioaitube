   312→function updateAssembleProgress(prev: AssembleProgressState, message: string, event?: AssembleEvent): AssembleProgressState {
   313→  if (!event) return { ...prev, message };
   314→  if (event.type === "log") {
   315→    return { ...prev, logs: [...prev.logs, event.message].slice(-20), message };
   316→  }
   317→  if (event.type === "engine") {
   318→    return { ...prev, message, phase: "Carregando engine", currentIndex: null, scenePct: 0, downloadItem: "engine", downloadPct: 0 };
   319→  }
   320→  if (event.type === "engine-download") {
   321→    return { ...prev, message, phase: "Carregando engine", currentIndex: null, scenePct: (event.pct / 100) * 5, downloadItem: `${event.mb}MB`, downloadPct: event.pct };
   322→  }
   323→  if (event.type === "scene-download") {
   324→    return { ...prev, message, phase: "Baixando recursos", currentIndex: event.index, total: event.total, downloadItem: event.item, downloadPct: event.pct, scenePct: 0 };
   325→  }
   326→  if (event.type === "scene-start") {
   327→    return { ...prev, message, phase: "Preparando cena", currentIndex: event.index, total: event.total, downloadItem: "", downloadPct: 0, scenePct: 10 };
   328→  }
   329→  if (event.type === "scene-render") {
   330→    return { ...prev, message, phase: "Renderizando cena", currentIndex: event.index, total: event.total, downloadItem: "", downloadPct: 0, scenePct: 30, ffmpegPct: 0 };
   331→  }
   332→  if (event.type === "ffmpeg-progress") {
   333→    const pct = Math.round((event.progress ?? 0) * 100);
   334→    return { ...prev, message, phase: "Renderizando cena", currentIndex: event.index, total: prev.total, scenePct: 30 + pct * 0.7, ffmpegPct: pct };
   335→  }
   336→  if (event.type === "scene-done") {
   337→    return { ...prev, message, phase: "Cena concluida", currentIndex: event.index, total: event.total, completed: Math.max(prev.completed, event.index + 1), downloadItem: "", downloadPct: 0, scenePct: 100 };
   338→  }
   339→  if (event.type === "concat") {
   340→    return { ...prev, message, phase: "Unindo cenas", currentIndex: null, total: event.total, completed: event.total, scenePct: 0 };
   341→  }
   342→  if (event.type === "done") {
   343→    return { ...prev, message, phase: "Video pronto", currentIndex: null, completed: prev.total, scenePct: 100 };
   344→  }
   345→  return { ...prev, message };

   348→/* --- Ken Burns Canvas Preview --- */
   349→function KenBurnsCanvas({ imageUrl, styleIndex, playing, vertical }: { imageUrl: string; styleIndex: number; playing: boolean; vertical?: boolean }) {
   350→  const canvasRef = useRef<HTMLCanvasElement>(null);
   351→  const imgRef = useRef<HTMLImageElement | null>(null);
   352→  const animRef = useRef<number>(0);
   353→  const startRef = useRef(0);
   354→
   355→  useEffect(() => {
   356→    const img = new Image();
   357→    img.crossOrigin = "anonymous";
   358→    img.src = imageUrl;
   359→    img.onload = () => { imgRef.current = img; };
   360→    imgRef.current = null;
   361→  }, [imageUrl]);
   362→
   363→  useEffect(() => {
   364→    const canvas = canvasRef.current;
   365→    if (!canvas) return;
   366→    const ctx = canvas.getContext("2d")!;
   367→    const W = canvas.width;
   368→    const H = canvas.height;
   369→    const DURATION = 6000;
   370→
   371→    function draw(ts: number) {
   372→      if (!startRef.current) startRef.current = ts;
   373→      const elapsed = ts - startRef.current;
   374→      const t = Math.min(elapsed / DURATION, 1);
   375→      const img = imgRef.current;
   376→
   377→      ctx.fillStyle = "#000";
   378→      ctx.fillRect(0, 0, W, H);
   379→
   380→      if (img) {
   381→        const zoom = 1.0 + t * 0.3;
   382→        const iw = W * zoom;
   383→        const ih = H * zoom;
   384→        let ox: number, oy: number;
   385→        switch (styleIndex % 5) {
   386→          case 0: ox = (W - iw) / 2; oy = (H - ih) / 2; break;
   387→          case 1: ox = (W - iw) / 2; oy = (H - ih) / 2; break;
   388→          case 2: ox = -(iw - W) * t; oy = (H - ih) / 2; break;
   389→          case 3: ox = -(iw - W) * (1 - t); oy = (H - ih) / 2; break;
   390→          default: ox = (W - iw) / 2; oy = -(ih - H) * t; break;
   391→        }
   392→        ctx.drawImage(img, ox, oy, iw, ih);
   393→        const grad = ctx.createRadialGradient(W / 2, H / 2, W * 0.25, W / 2, H / 2, W * 0.7);
   394→        grad.addColorStop(0, "rgba(0,0,0,0)");
   395→        grad.addColorStop(1, "rgba(0,0,0,0.6)");
   396→        ctx.fillStyle = grad;
   397→        ctx.fillRect(0, 0, W, H);
   398→      }
   399→
   400→      // Cinematic bars only for horizontal
   401→      if (!vertical) {
   402→        ctx.fillStyle = "rgba(0,0,0,0.5)";
   403→        ctx.fillRect(0, 0, W, H * 0.07);
   404→        ctx.fillRect(0, H * 0.93, W, H * 0.07);
   405→      }
   406→
   407→      const scanY = (elapsed * 0.05) % H;
   408→      ctx.fillStyle = "rgba(139,92,246,0.08)";
   409→      ctx.fillRect(0, scanY, W, 2);
   410→
   411→      if (playing && t < 1) {
   412→        animRef.current = requestAnimationFrame(draw);
   413→      } else if (playing && t >= 1) {
   414→        startRef.current = 0;
   415→        animRef.current = requestAnimationFrame(draw);
   416→      }
   417→    }
   418→
   419→    if (playing) {
   420→      startRef.current = 0;
   421→      animRef.current = requestAnimationFrame(draw);
   422→    }
   423→    return () => cancelAnimationFrame(animRef.current);
   424→  }, [playing, styleIndex, imageUrl, vertical]);
   425→
   426→  return (
   427→    <canvas
   428→      ref={canvasRef}
   429→      width={vertical ? 180 : 640}
   430→      height={vertical ? 320 : 360}
   431→      className="w-full h-full object-cover rounded-xl"
   432→    />
   433→  );
   434→}

   436→/* --- Audio Waveform Visualizer --- */
   437→function AudioWaveform({ playing, color = "#a78bfa" }: { playing: boolean; color?: string }) {
   438→  const canvasRef = useRef<HTMLCanvasElement>(null);
   439→  const animRef = useRef<number>(0);
   440→
   441→  useEffect(() => {
   442→    const canvas = canvasRef.current;
   443→    if (!canvas) return;
   444→    const ctx = canvas.getContext("2d")!;
   445→    const W = canvas.width;
   446→    const H = canvas.height;
   447→    const BARS = 48;
   448→    const barW = W / BARS;
   449→    let phase = 0;
   450→
   451→    function draw() {
   452→      ctx.clearRect(0, 0, W, H);
   453→      phase += 0.04;
   454→
   455→      for (let i = 0; i < BARS; i++) {
   456→        const x = i * barW;
   457→        let h: number;
   458→        if (playing) {
   459→          const wave = Math.sin(phase + i * 0.3) * 0.4 + Math.sin(phase * 1.5 + i * 0.15) * 0.3 + Math.cos(phase * 0.7 + i * 0.5) * 0.3;
   460→          h = Math.abs(wave) * H * 0.85 + H * 0.08;
   461→        } else {
   462→          h = H * 0.1;
   463→        }
   464→        const grad = ctx.createLinearGradient(x, H - h, x, H);
   465→        grad.addColorStop(0, color);
   466→        grad.addColorStop(1, color + "33");
   467→        ctx.fillStyle = grad;
   468→        ctx.fillRect(x + 1, H - h, barW - 2, h);
   469→      }
   470→      animRef.current = requestAnimationFrame(draw);
   471→    }
   472→    animRef.current = requestAnimationFrame(draw);
   473→    return () => cancelAnimationFrame(animRef.current);
   474→  }, [playing, color]);
   475→
   476→  return (
   477→    <canvas ref={canvasRef} width={384} height={48} className="w-full h-12 rounded-lg" />
   478→  );
   479→}

   481→/* --- Floating Particles --- */
   482→function Particles({ count = 20 }: { count?: number }) {
   483→  const canvasRef = useRef<HTMLCanvasElement>(null);
   484→  const animRef = useRef<number>(0);
   485→  const particlesRef = useRef<{ x: number; y: number; vx: number; vy: number; r: number; a: number }[]>([]);
   486→
   487→  useEffect(() => {
   488→    const canvas = canvasRef.current;
   489→    if (!canvas) return;
   490→    const ctx = canvas.getContext("2d")!;
   491→    const W = canvas.width;
   492→    const H = canvas.height;
   493→    particlesRef.current = Array.from({ length: count }, () => ({
   494→      x: Math.random() * W, y: Math.random() * H,
   495→      vx: (Math.random() - 0.5) * 0.5, vy: -Math.random() * 0.8 - 0.2,
   496→      r: Math.random() * 2 + 0.5, a: Math.random() * 0.5 + 0.2,
   497→    }));
   498→
   499→    function draw() {
   500→      ctx.clearRect(0, 0, W, H);
   501→      for (const p of particlesRef.current) {
   502→        p.x += p.vx; p.y += p.vy;
   503→        if (p.y < -5) { p.y = H + 5; p.x = Math.random() * W; }
   504→        if (p.x < -5) p.x = W + 5;
   505→        if (p.x > W + 5) p.x = -5;
   506→        ctx.beginPath();
   507→        ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
   508→        ctx.fillStyle = `rgba(167,139,250,${p.a})`;
   509→        ctx.fill();
   510→      }
   511→      animRef.current = requestAnimationFrame(draw);
   512→    }
   513→    animRef.current = requestAnimationFrame(draw);
   514→    return () => cancelAnimationFrame(animRef.current);
   515→  }, [count]);
   516→
   517→  return <canvas ref={canvasRef} width={200} height={200} className="absolute inset-0 w-full h-full pointer-events-none" />;
   518→}

   884→  function keySceneIndexes(total: number): number[] {
   885→    if (total <= 0) return [];
   886→    if (total <= 3) return Array.from({ length: total }, (_, i) => i);
   887→    const count = Math.max(2, Math.ceil(total * 0.2));
   888→    const set = new Set<number>([0, total - 1]);
   889→    const middleSlots = count - 2;
   890→    for (let k = 1; k <= middleSlots; k++) {
   891→      set.add(Math.round((k * (total - 1)) / (middleSlots + 1)));
   892→    }
   893→    return [...set].sort((a, b) => a - b);
   894→  }

   912→  async function genThumbnail() {

   922→      const fontFam = brand.fontStyle === "serif"
   923→        ? "'Georgia', 'Times New Roman', serif"

   925→          ? "'Bebas Neue', 'Oswald', 'Impact', sans-serif"
   926→          : "'Impact', 'Arial Black', sans-serif";

   928→      const prompt = (script.thumbnailPrompt || `dark dramatic YouTube thumbnail about ${script.title}, cinematic, high contrast, no text, 16:9`) + `, ${mood}`;
   929→      let baseUrl = "";
   930→      const thumbResult = await clientImage(prompt);
   931→      baseUrl = `data:${thumbResult.mime};base64,${thumbResult.b64}`;
   932→      if (!baseUrl) throw new Error("Sem imagem base");
   933→
   934→      const img = new Image();
   935→      img.crossOrigin = "anonymous";
   936→      await new Promise<void>((res, rej) => { img.onload = () => res(); img.onerror = () => rej(new Error("Falha ao carregar imagem")); img.src = baseUrl; });
   937→
   938→      const canvas = document.createElement("canvas");
   939→      canvas.width = 1280; canvas.height = 720;
   940→      const ctx = canvas.getContext("2d")!;
   941→      const scale = Math.max(1280 / img.width, 720 / img.height);
   942→      const dw = img.width * scale, dh = img.height * scale;
   943→      ctx.drawImage(img, (1280 - dw) / 2, (720 - dh) / 2, dw, dh);
   944→      const grad = ctx.createLinearGradient(0, 300, 0, 720);
   945→      grad.addColorStop(0, "rgba(0,0,0,0)");
   946→      grad.addColorStop(1, "rgba(0,0,0,0.85)");

   970→        ctx.fillStyle = fillColor;

  1607→      </div>

  1610→        <p>

  1623→/* ── Publish to TikTok ────────────────────────────────────────── */
  1624→
  1625→function PublishToTikTok({ videoUrl, script }: { videoUrl: string; script: Script }) {
  1626→  const [open, setOpen] = useState(false);
  1627→  const [sending, setSending] = useState(false);
  1628→  const [status, setStatus] = useState("");
  1629→  const [privacy, setPrivacy] = useState<"PUBLIC_TO_EVERYONE" | "FOLLOWERS_ONLY" | "PRIVATE">("PUBLIC_TO_EVERYONE");
  1630→  const [connected, setConnected] = useState(() => !!localStorage.getItem("aidc.tiktok_access_token"));
  1631→  const [title, setTitle] = useState(() => (script.seoTitle || script.title).slice(0, 150));
  1632→
  1633→  useEffect(() => {
  1634→    // Listen for OAuth callback messages
  1635→    function handleMessage(e: MessageEvent) {
  1636→      if (e.data?.type === "tiktok-auth" && e.data.code) {
  1637→        // Exchange code for token
  1638→        fetch("/api/tiktok", {
  1639→          method: "POST",
  1640→          headers: { "Content-Type": "application/json" },
  1641→          body: JSON.stringify({ action: "token", code: e.data.code }),
  1642→        })
  1643→          .then((r) => r.json())
  1644→          .then((data) => {
  1645→            if (data.accessToken) {
  1646→              localStorage.setItem("aidc.tiktok_access_token", data.accessToken);
  1647→              setConnected(true);
  1648→              toast.success("TikTok conectado com sucesso!");
  1649→            } else {
  1650→              toast.error(`Erro TikTok: ${data.error}`);
  1651→            }
  1652→          })
  1653→          .catch((err) => toast.error(`Erro: ${err.message}`));
  1654→      }
  1655→    }
  1656→    window.addEventListener("message", handleMessage);
  1657→    return () => window.removeEventListener("message", handleMessage);
  1658→  }, []);
  1659→
  1660→  async function startOAuth() {
  1661→    try {
  1662→      const res = await fetch("/api/tiktok", {
  1663→        method: "POST",
  1664→        headers: { "Content-Type": "application/json" },
  1665→        body: JSON.stringify({ action: "auth-url" }),
  1666→      });
  1667→      const data = await res.json();
  1668→      if (data.authUrl) {
  1669→        window.open(data.authUrl, "tiktok-auth", "width=600,height=700");
  1670→        toast.info("Autorize o acesso na janela do TikTok...");
  1671→      } else {
  1672→        toast.error(data.error || "Erro ao iniciar OAuth");
  1673→      }
  1674→    } catch (err: any) {
  1675→      toast.error(`Erro: ${err.message}`);
  1676→    }
  1677→  }
  1678→
  1679→  async function publish() {
  1680→    const accessToken = localStorage.getItem("aidc.tiktok_access_token");
  1681→    if (!accessToken) {
  1682→      toast.error("Conecte sua conta TikTok primeiro.");
  1683→      return;
  1684→    }
  1685→
  1686→    setSending(true);
  1687→    setStatus("Hospedando video temporariamente...");
  1688→    try {
  1689→      // Upload to litterbox for TikTok to pull
  1690→      const videoBlob = await (await fetch(videoUrl)).blob();
  1691→      const up = new FormData();
  1692→      up.append("reqtype", "fileupload");
  1693→      up.append("time", "24h");
  1694→      up.append("fileToUpload", videoBlob, "tiktok-video.webm");
  1695→      const upRes = await fetch("https://litterbox.catbox.moe/resources/internals/api.php", { method: "POST", body: up });
  1696→      const hostedUrl = (await upRes.text()).trim();
  1697→      if (!upRes.ok || !hostedUrl.startsWith("http")) throw new Error("Falha ao hospedar video");
  1698→
  1699→      setStatus("Enviando ao TikTok...");
  1700→      const res = await fetch("/api/tiktok", {
  1701→        method: "POST",
  1702→        headers: { "Content-Type": "application/json" },
  1703→        body: JSON.stringify({
  1704→          action: "publish",
  1705→          videoUrl: hostedUrl,
  1706→          title: title.slice(0, 150),
  1707→          privacyLevel: privacy,
  1708→          accessToken,
  1709→        }),
  1710→      });
  1711→      const data = await res.json();
  1712→      if (data.ok) {
  1713→        toast.success("Video enviado ao TikTok! Pode levar alguns minutos para aparecer.");
  1714→        setOpen(false);
  1715→      } else {
  1716→        toast.error(data.error || "Erro ao publicar no TikTok", { duration: 9000 });
  1717→      }
  1718→    } catch (err: any) {
  1719→      toast.error(err?.message ?? "Falha ao publicar no TikTok", { duration: 9000 });
  1720→    } finally {
  1721→      setSending(false);
  1722→      setStatus("");
  1723→    }
  1724→  }
  1725→
  1726→  if (!open) {
  1727→    return (
  1728→      <button
  1729→        onClick={() => setOpen(true)}
  1730→        className="inline-block rounded-md border border-[#00f2ea]/30 bg-[#00f2ea]/10 hover:bg-[#00f2ea]/20 px-4 py-2 text-sm font-medium text-[#00f2ea]"
  1731→      >
  1732→        Publicar no TikTok
  1733→      </button>
  1734→    );
  1735→  }
  1736→
  1737→  return (
  1738→    <div className="w-full rounded-md border border-[#00f2ea]/20 bg-[#00f2ea]/5 p-3 space-y-3">
  1739→      <div className="flex items-center gap-2">
  1740→        <Smartphone className="w-4 h-4 text-[#00f2ea]" />
  1741→        <span className="text-sm font-bold text-[#00f2ea]">Publicar no TikTok</span>
  1742→      </div>
  1743→
  1744→      {!connected ? (
  1745→        <div className="space-y-3">
  1746→          <p className="text-xs text-muted-foreground">
  1747→            Conecte sua conta TikTok para publicar videos diretamente. Voce sera redirecionado para autorizar o acesso.
  1748→          </p>
  1749→          <button
  1750→            onClick={startOAuth}
  1751→            className="w-full rounded-md bg-[#00f2ea] text-black font-semibold px-4 py-2.5 text-sm hover:bg-[#00f2ea]/90 transition"
  1752→          >
  1753→            Conectar conta TikTok
  1754→          </button>
  1755→        </div>
  1756→      ) : (
  1757→        <>
  1758→          <div className="flex items-center gap-2">
  1759→            <span className="w-2 h-2 rounded-full bg-green-400" />

  1761→          </div>
  1762→
  1763→          <div className="space-y-1.5">
  1764→            <label className="text-xs text-muted-foreground">Titulo do video (max 150 chars)</label>
  1765→            <input
  1766→              type="text"
  1767→              value={title}
  1768→              onChange={(e) => setTitle(e.target.value.slice(0, 150))}
  1769→              className="w-full rounded-md bg-background border border-white/10 px-3 py-2 text-sm"
  1770→              maxLength={150}
  1771→            />

  1773→          </div>
  1774→
  1775→          <div className="space-y-1.5">

  1798→
  1799→          <div className="flex flex-wrap gap-2">
  1800→            <button onClick={publish} disabled={sending} className="rounded-md bg-[#00f2ea] text-black font-semibold px-4 py-2 text-sm hover:bg-[#00f2ea]/90 transition disabled:opacity-50">
  1801→              {sending ? "Enviando..." : "Publicar no TikTok"}
