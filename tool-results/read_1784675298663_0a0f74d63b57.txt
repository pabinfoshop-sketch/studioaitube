   333‚Üí   333‚Üí   333‚Üífunction KenBurnsCanvas({ imageUrl, styleIndex, playing }: { imageUrl: string; styleIndex: number; playing: boolean }) {
   334‚Üí   334‚Üí   334‚Üí  const canvasRef = useRef<HTMLCanvasElement>(null);
   335‚Üí   335‚Üí   335‚Üí  const imgRef = useRef<HTMLImageElement | null>(null);
   336‚Üí   336‚Üí   336‚Üí  const animRef = useRef<number>(0);
   337‚Üí   337‚Üí   337‚Üí  const startRef = useRef(0);
   338‚Üí   338‚Üí   338‚Üí
   339‚Üí   339‚Üí   339‚Üí  useEffect(() => {
   340‚Üí   340‚Üí   340‚Üí    const img = new Image();
   341‚Üí   341‚Üí   341‚Üí    img.crossOrigin = "anonymous";
   342‚Üí   342‚Üí   342‚Üí    img.src = imageUrl;
   343‚Üí   343‚Üí   343‚Üí    img.onload = () => { imgRef.current = img; };
   344‚Üí   344‚Üí   344‚Üí    imgRef.current = null;
   345‚Üí   345‚Üí   345‚Üí  }, [imageUrl]);
   346‚Üí   346‚Üí   346‚Üí
   347‚Üí   347‚Üí   347‚Üí  useEffect(() => {
   348‚Üí   348‚Üí   348‚Üí    const canvas = canvasRef.current;
   349‚Üí   349‚Üí   349‚Üí    if (!canvas) return;
   350‚Üí   350‚Üí   350‚Üí    const ctx = canvas.getContext("2d")!;
   351‚Üí   351‚Üí   351‚Üí    const W = canvas.width;

   355‚Üí   355‚Üí   355‚Üí    function draw(ts: number) {
   356‚Üí   356‚Üí   356‚Üí      if (!startRef.current) startRef.current = ts;
   357‚Üí   357‚Üí   357‚Üí      const elapsed = ts - startRef.current;
   358‚Üí   358‚Üí   358‚Üí      const t = Math.min(elapsed / DURATION, 1);
   359‚Üí   359‚Üí   359‚Üí      const img = imgRef.current;
   360‚Üí   360‚Üí   360‚Üí

   363‚Üí   363‚Üí   363‚Üí
   364‚Üí   364‚Üí   364‚Üí      if (img) {
   365‚Üí   365‚Üí   365‚Üí        const zoom = 1.0 + t * 0.3;
   366‚Üí   366‚Üí   366‚Üí        const iw = W * zoom;
   367‚Üí   367‚Üí   367‚Üí        const ih = H * zoom;
   368‚Üí   368‚Üí   368‚Üí        let ox: number, oy: number;
   369‚Üí   369‚Üí   369‚Üí        switch (styleIndex % 5) {
   370‚Üí   370‚Üí   370‚Üí          case 0: ox = (W - iw) / 2; oy = (H - ih) / 2; break;
   371‚Üí   371‚Üí   371‚Üí          case 1: ox = (W - iw) / 2; oy = (H - ih) / 2; break;
   372‚Üí   372‚Üí   372‚Üí          case 2: ox = -(iw - W) * t; oy = (H - ih) / 2; break;
   373‚Üí   373‚Üí   373‚Üí          case 3: ox = -(iw - W) * (1 - t); oy = (H - ih) / 2; break;
   374‚Üí   374‚Üí   374‚Üí          default: ox = (W - iw) / 2; oy = -(ih - H) * t; break;
   375‚Üí   375‚Üí   375‚Üí        }

   377‚Üí   377‚Üí   377‚Üí        // vignette

   384‚Üí   384‚Üí   384‚Üí
   385‚Üí   385‚Üí   385‚Üí      // cinematic bars
   386‚Üí   386‚Üí   386‚Üí      ctx.fillStyle = "rgba(0,0,0,0.5)";
   387‚Üí   387‚Üí   387‚Üí      ctx.fillRect(0, 0, W, H * 0.07);
   388‚Üí   388‚Üí   388‚Üí      ctx.fillRect(0, H * 0.93, W, H * 0.07);
   389‚Üí   389‚Üí   389‚Üí
   390‚Üí   390‚Üí   390‚Üí      // scanline
   391‚Üí   391‚Üí   391‚Üí      const scanY = (elapsed * 0.05) % H;
   392‚Üí   392‚Üí   392‚Üí      ctx.fillStyle = "rgba(139,92,246,0.08)";
   393‚Üí   393‚Üí   393‚Üí      ctx.fillRect(0, scanY, W, 2);
   394‚Üí   394‚Üí   394‚Üí
