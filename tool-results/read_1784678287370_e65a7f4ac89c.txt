    67→function useCountUp(target: number, duration = 1600, start = false) {
    68→  const [n, setN] = useState(0);
    69→  useEffect(() => {
    70→    if (!start) return;
    71→    const t0 = performance.now();
    72→    let raf = 0;
    73→    const tick = (t: number) => {
    74→      const p = Math.min(1, (t - t0) / duration);
    75→      const eased = 1 - Math.pow(1 - p, 3);
    76→      setN(Math.floor(eased * target));
    77→      if (p < 1) raf = requestAnimationFrame(tick);
    78→    };
    79→    raf = requestAnimationFrame(tick);
    80→    return () => cancelAnimationFrame(raf);
    81→  }, [target, duration, start]);
    82→  return n;
    83→}
