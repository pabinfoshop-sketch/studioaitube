   295→   295→function updateAssembleProgress(prev: AssembleProgressState, message: string, event?: AssembleEvent): AssembleProgressState {
   296→   296→  if (!event) return { ...prev, message };
   297→   297→  if (event.type === "log") {
   298→   298→    return { ...prev, logs: [...prev.logs, event.message].slice(-20), message };
   299→   299→  }
   300→   300→  if (event.type === "engine") {
   301→   301→    return { ...prev, message, phase: "Carregando engine", currentIndex: null, scenePct: 0, downloadItem: "engine", downloadPct: 0 };
   302→   302→  }
   303→   303→  if (event.type === "engine-download") {
   304→   304→    // Show engine download as 0-5% of total progress
   305→   305→    return { ...prev, message, phase: "Carregando engine", currentIndex: null, scenePct: (event.pct / 100) * 5, downloadItem: `${event.mb}MB`, downloadPct: event.pct };
   306→   306→  }
   307→   307→  if (event.type === "scene-download") {
   308→   308→    return { ...prev, message, phase: "Baixando recursos", currentIndex: event.index, total: event.total, downloadItem: event.item, downloadPct: event.pct, scenePct: 0 };
   309→   309→  }
   310→   310→  if (event.type === "scene-start") {
   311→   311→    return { ...prev, message, phase: "Preparando cena", currentIndex: event.index, total: event.total, downloadItem: "", downloadPct: 0, scenePct: 10 };
   312→   312→  }
   313→   313→  if (event.type === "scene-render") {
   314→   314→    return { ...prev, message, phase: "Renderizando cena", currentIndex: event.index, total: event.total, downloadItem: "", downloadPct: 0, scenePct: 30, ffmpegPct: 0 };
   315→   315→  }
   316→   316→  if (event.type === "ffmpeg-progress") {
   317→   317→    const pct = Math.round((event.progress ?? 0) * 100);
   318→   318→    return { ...prev, message, phase: "Renderizando cena", currentIndex: event.index, total: prev.total, scenePct: 30 + pct * 0.7, ffmpegPct: pct };
   319→   319→  }
   320→   320→  if (event.type === "scene-done") {
   321→   321→    return { ...prev, message, phase: "Cena concluida", currentIndex: event.index, total: event.total, completed: Math.max(prev.completed, event.index + 1), downloadItem: "", downloadPct: 0, scenePct: 100 };
   322→   322→  }
   323→   323→  if (event.type === "concat") {
   324→   324→    return { ...prev, message, phase: "Unindo cenas", currentIndex: null, total: event.total, completed: event.total, scenePct: 0 };
   325→   325→  }
   326→   326→  if (event.type === "done") {
   327→   327→    return { ...prev, message, phase: "Video pronto", currentIndex: null, completed: prev.total, scenePct: 100 };
   328→   328→  }
   329→   329→  return { ...prev, message };
   330→   330→}

   332→   332→/* --- Ken Burns Canvas Preview --- */
   333→   333→function KenBurnsCanvas({ imageUrl, styleIndex, playing }: { imageUrl: string; styleIndex: number; playing: boolean }) {
   334→   334→  const canvasRef = useRef<HTMLCanvasElement>(null);
   335→   335→  const imgRef = useRef<HTMLImageElement | null>(null);
   336→   336→  const animRef = useRef<number>(0);
   337→   337→  const startRef = useRef(0);
   338→   338→
   339→   339→  useEffect(() => {
   340→   340→    const img = new Image();
   341→   341→    img.crossOrigin = "anonymous";
   342→   342→    img.src = imageUrl;
   343→   343→    img.onload = () => { imgRef.current = img; };
   344→   344→    imgRef.current = null;
   345→   345→  }, [imageUrl]);
   346→   346→
   347→   347→  useEffect(() => {
   348→   348→    const canvas = canvasRef.current;
   349→   349→    if (!canvas) return;
   350→   350→    const ctx = canvas.getContext("2d")!;
   351→   351→    const W = canvas.width;
   352→   352→    const H = canvas.height;
   353→   353→    const DURATION = 6000;
   354→   354→
   355→   355→    function draw(ts: number) {
   356→   356→      if (!startRef.current) startRef.current = ts;
   357→   357→      const elapsed = ts - startRef.current;
   358→   358→      const t = Math.min(elapsed / DURATION, 1);
   359→   359→      const img = imgRef.current;
   360→   360→
   361→   361→      ctx.fillStyle = "#000";
   362→   362→      ctx.fillRect(0, 0, W, H);
   363→   363→
   364→   364→      if (img) {
   365→   365→        const zoom = 1.0 + t * 0.3;
   366→   366→        const iw = W * zoom;
   367→   367→        const ih = H * zoom;
   368→   368→        let ox: number, oy: number;
   369→   369→        switch (styleIndex % 5) {
   370→   370→          case 0: ox = (W - iw) / 2; oy = (H - ih) / 2; break;
   371→   371→          case 1: ox = (W - iw) / 2; oy = (H - ih) / 2; break;
   372→   372→          case 2: ox = -(iw - W) * t; oy = (H - ih) / 2; break;
   373→   373→          case 3: ox = -(iw - W) * (1 - t); oy = (H - ih) / 2; break;
   374→   374→          default: ox = (W - iw) / 2; oy = -(ih - H) * t; break;
   375→   375→        }
   376→   376→        ctx.drawImage(img, ox, oy, iw, ih);
   377→   377→        // vignette
   378→   378→        const grad = ctx.createRadialGradient(W / 2, H / 2, W * 0.25, W / 2, H / 2, W * 0.7);
   379→   379→        grad.addColorStop(0, "rgba(0,0,0,0)");
   380→   380→        grad.addColorStop(1, "rgba(0,0,0,0.6)");
   381→   381→        ctx.fillStyle = grad;
   382→   382→        ctx.fillRect(0, 0, W, H);
   383→   383→      }
   384→   384→
   385→   385→      // cinematic bars
   386→   386→      ctx.fillStyle = "rgba(0,0,0,0.5)";
   387→   387→      ctx.fillRect(0, 0, W, H * 0.07);
   388→   388→      ctx.fillRect(0, H * 0.93, W, H * 0.07);
   389→   389→
   390→   390→      // scanline
   391→   391→      const scanY = (elapsed * 0.05) % H;
   392→   392→      ctx.fillStyle = "rgba(139,92,246,0.08)";
   393→   393→      ctx.fillRect(0, scanY, W, 2);
   394→   394→
   395→   395→      if (playing && t < 1) {
   396→   396→        animRef.current = requestAnimationFrame(draw);
   397→   397→      } else if (playing && t >= 1) {
   398→   398→        startRef.current = 0;
   399→   399→        animRef.current = requestAnimationFrame(draw);
   400→   400→      }
   401→   401→    }
   402→   402→
   403→   403→    if (playing) {
   404→   404→      startRef.current = 0;
   405→   405→      animRef.current = requestAnimationFrame(draw);
   406→   406→    }
   407→   407→    return () => cancelAnimationFrame(animRef.current);
   408→   408→  }, [playing, styleIndex, imageUrl]);
   409→   409→
   410→   410→  return (
   411→   411→    <canvas
   412→   412→      ref={canvasRef}
   413→   413→      width={640}
   414→   414→      height={360}
   415→   415→      className="w-full h-full object-cover rounded-xl"
   416→   416→    />
   417→   417→  );
   418→   418→}

   420→   420→/* --- Audio Waveform Visualizer --- */
   421→   421→function AudioWaveform({ playing, color = "#a78bfa" }: { playing: boolean; color?: string }) {
   422→   422→  const canvasRef = useRef<HTMLCanvasElement>(null);
   423→   423→  const animRef = useRef<number>(0);
   424→   424→
   425→   425→  useEffect(() => {
   426→   426→    const canvas = canvasRef.current;
   427→   427→    if (!canvas) return;
   428→   428→    const ctx = canvas.getContext("2d")!;
   429→   429→    const W = canvas.width;
   430→   430→    const H = canvas.height;
   431→   431→    const BARS = 48;
   432→   432→    const barW = W / BARS;
   433→   433→    let phase = 0;
   434→   434→
   435→   435→    function draw() {
   436→   436→      ctx.clearRect(0, 0, W, H);
   437→   437→      phase += 0.04;
   438→   438→
   439→   439→      for (let i = 0; i < BARS; i++) {
   440→   440→        const x = i * barW;
   441→   441→        let h: number;
   442→   442→        if (playing) {
   443→   443→          const wave = Math.sin(phase + i * 0.3) * 0.4 + Math.sin(phase * 1.5 + i * 0.15) * 0.3 + Math.cos(phase * 0.7 + i * 0.5) * 0.3;
   444→   444→          h = Math.abs(wave) * H * 0.85 + H * 0.08;
   445→   445→        } else {
   446→   446→          h = H * 0.1;
   447→   447→        }
   448→   448→        const grad = ctx.createLinearGradient(x, H - h, x, H);
   449→   449→        grad.addColorStop(0, color);
   450→   450→        grad.addColorStop(1, color + "33");
   451→   451→        ctx.fillStyle = grad;
   452→   452→        ctx.fillRect(x + 1, H - h, barW - 2, h);
   453→   453→      }
   454→   454→      animRef.current = requestAnimationFrame(draw);
   455→   455→    }
   456→   456→    animRef.current = requestAnimationFrame(draw);
   457→   457→    return () => cancelAnimationFrame(animRef.current);
   458→   458→  }, [playing, color]);
   459→   459→
   460→   460→  return (
   461→   461→    <canvas ref={canvasRef} width={384} height={48} className="w-full h-12 rounded-lg" />
   462→   462→  );
   463→   463→}

   465→   465→/* --- Floating Particles --- */
   466→   466→function Particles({ count = 20 }: { count?: number }) {
   467→   467→  const canvasRef = useRef<HTMLCanvasElement>(null);
   468→   468→  const animRef = useRef<number>(0);
   469→   469→  const particlesRef = useRef<{ x: number; y: number; vx: number; vy: number; r: number; a: number }[]>([]);
   470→   470→
   471→   471→  useEffect(() => {
   472→   472→    const canvas = canvasRef.current;
   473→   473→    if (!canvas) return;
   474→   474→    const ctx = canvas.getContext("2d")!;
   475→   475→    const W = canvas.width;
   476→   476→    const H = canvas.height;
   477→   477→    particlesRef.current = Array.from({ length: count }, () => ({
   478→   478→      x: Math.random() * W, y: Math.random() * H,
   479→   479→      vx: (Math.random() - 0.5) * 0.5, vy: -Math.random() * 0.8 - 0.2,
   480→   480→      r: Math.random() * 2 + 0.5, a: Math.random() * 0.5 + 0.2,
   481→   481→    }));
   482→   482→
   483→   483→    function draw() {
   484→   484→      ctx.clearRect(0, 0, W, H);
   485→   485→      for (const p of particlesRef.current) {
   486→   486→        p.x += p.vx; p.y += p.vy;
   487→   487→        if (p.y < -5) { p.y = H + 5; p.x = Math.random() * W; }
   488→   488→        if (p.x < -5) p.x = W + 5;
   489→   489→        if (p.x > W + 5) p.x = -5;
   490→   490→        ctx.beginPath();
   491→   491→        ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
   492→   492→        ctx.fillStyle = `rgba(167,139,250,${p.a})`;
   493→   493→        ctx.fill();
   494→   494→      }
   495→   495→      animRef.current = requestAnimationFrame(draw);
   496→   496→    }
   497→   497→    animRef.current = requestAnimationFrame(draw);
   498→   498→    return () => cancelAnimationFrame(animRef.current);
   499→   499→  }, [count]);
   500→   500→
   501→   501→  return <canvas ref={canvasRef} width={200} height={200} className="absolute inset-0 w-full h-full pointer-events-none" />;
   502→   502→}

   523→   523→        if (Math.abs(diff) < 0.5) return raw;
   524→   524→        return prev + diff * 0.08;
