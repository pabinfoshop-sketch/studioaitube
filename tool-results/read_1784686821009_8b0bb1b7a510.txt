   301→   301→function updateAssembleProgress(prev: AssembleProgressState, message: string, event?: AssembleEvent): AssembleProgressState {
   302→   302→  if (!event) return { ...prev, message };
   303→   303→  if (event.type === "log") {
   304→   304→    return { ...prev, logs: [...prev.logs, event.message].slice(-20), message };
   305→   305→  }
   306→   306→  if (event.type === "engine") {
   307→   307→    return { ...prev, message, phase: "Carregando engine", currentIndex: null, scenePct: 0, downloadItem: "engine", downloadPct: 0 };
   308→   308→  }
   309→   309→  if (event.type === "engine-download") {
   310→   310→    // Show engine download as 0-5% of total progress
   311→   311→    return { ...prev, message, phase: "Carregando engine", currentIndex: null, scenePct: (event.pct / 100) * 5, downloadItem: `${event.mb}MB`, downloadPct: event.pct };
   312→   312→  }
   313→   313→  if (event.type === "scene-download") {
   314→   314→    return { ...prev, message, phase: "Baixando recursos", currentIndex: event.index, total: event.total, downloadItem: event.item, downloadPct: event.pct, scenePct: 0 };
   315→   315→  }
   316→   316→  if (event.type === "scene-start") {
   317→   317→    return { ...prev, message, phase: "Preparando cena", currentIndex: event.index, total: event.total, downloadItem: "", downloadPct: 0, scenePct: 10 };
   318→   318→  }
   319→   319→  if (event.type === "scene-render") {
   320→   320→    return { ...prev, message, phase: "Renderizando cena", currentIndex: event.index, total: event.total, downloadItem: "", downloadPct: 0, scenePct: 30, ffmpegPct: 0 };
   321→   321→  }
   322→   322→  if (event.type === "ffmpeg-progress") {
   323→   323→    const pct = Math.round((event.progress ?? 0) * 100);
   324→   324→    return { ...prev, message, phase: "Renderizando cena", currentIndex: event.index, total: prev.total, scenePct: 30 + pct * 0.7, ffmpegPct: pct };
   325→   325→  }
   326→   326→  if (event.type === "scene-done") {
   327→   327→    return { ...prev, message, phase: "Cena concluida", currentIndex: event.index, total: event.total, completed: Math.max(prev.completed, event.index + 1), downloadItem: "", downloadPct: 0, scenePct: 100 };
   328→   328→  }
   329→   329→  if (event.type === "concat") {
   330→   330→    return { ...prev, message, phase: "Unindo cenas", currentIndex: null, total: event.total, completed: event.total, scenePct: 0 };
   331→   331→  }
   332→   332→  if (event.type === "done") {
   333→   333→    return { ...prev, message, phase: "Video pronto", currentIndex: null, completed: prev.total, scenePct: 100 };
   334→   334→  }
   335→   335→  return { ...prev, message };
   336→   336→}

   338→   338→/* --- Ken Burns Canvas Preview --- */
   339→   339→function KenBurnsCanvas({ imageUrl, styleIndex, playing }: { imageUrl: string; styleIndex: number; playing: boolean }) {
   340→   340→  const canvasRef = useRef<HTMLCanvasElement>(null);
   341→   341→  const imgRef = useRef<HTMLImageElement | null>(null);
   342→   342→  const animRef = useRef<number>(0);
   343→   343→  const startRef = useRef(0);
   344→   344→
   345→   345→  useEffect(() => {
   346→   346→    const img = new Image();
   347→   347→    img.crossOrigin = "anonymous";
   348→   348→    img.src = imageUrl;
   349→   349→    img.onload = () => { imgRef.current = img; };
   350→   350→    imgRef.current = null;
   351→   351→  }, [imageUrl]);
   352→   352→
   353→   353→  useEffect(() => {
   354→   354→    const canvas = canvasRef.current;
   355→   355→    if (!canvas) return;
   356→   356→    const ctx = canvas.getContext("2d")!;
   357→   357→    const W = canvas.width;
   358→   358→    const H = canvas.height;
   359→   359→    const DURATION = 6000;
   360→   360→
   361→   361→    function draw(ts: number) {
   362→   362→      if (!startRef.current) startRef.current = ts;
   363→   363→      const elapsed = ts - startRef.current;
   364→   364→      const t = Math.min(elapsed / DURATION, 1);
   365→   365→      const img = imgRef.current;
   366→   366→
   367→   367→      ctx.fillStyle = "#000";
   368→   368→      ctx.fillRect(0, 0, W, H);
   369→   369→
   370→   370→      if (img) {
   371→   371→        const zoom = 1.0 + t * 0.3;
   372→   372→        const iw = W * zoom;
   373→   373→        const ih = H * zoom;
   374→   374→        let ox: number, oy: number;
   375→   375→        switch (styleIndex % 5) {
   376→   376→          case 0: ox = (W - iw) / 2; oy = (H - ih) / 2; break;
   377→   377→          case 1: ox = (W - iw) / 2; oy = (H - ih) / 2; break;
   378→   378→          case 2: ox = -(iw - W) * t; oy = (H - ih) / 2; break;
   379→   379→          case 3: ox = -(iw - W) * (1 - t); oy = (H - ih) / 2; break;
   380→   380→          default: ox = (W - iw) / 2; oy = -(ih - H) * t; break;
   381→   381→        }
   382→   382→        ctx.drawImage(img, ox, oy, iw, ih);
   383→   383→        // vignette
   384→   384→        const grad = ctx.createRadialGradient(W / 2, H / 2, W * 0.25, W / 2, H / 2, W * 0.7);
   385→   385→        grad.addColorStop(0, "rgba(0,0,0,0)");
   386→   386→        grad.addColorStop(1, "rgba(0,0,0,0.6)");
   387→   387→        ctx.fillStyle = grad;
   388→   388→        ctx.fillRect(0, 0, W, H);
   389→   389→      }
   390→   390→
   391→   391→      // cinematic bars
   392→   392→      ctx.fillStyle = "rgba(0,0,0,0.5)";
   393→   393→      ctx.fillRect(0, 0, W, H * 0.07);
   394→   394→      ctx.fillRect(0, H * 0.93, W, H * 0.07);
   395→   395→
   396→   396→      // scanline
   397→   397→      const scanY = (elapsed * 0.05) % H;
   398→   398→      ctx.fillStyle = "rgba(139,92,246,0.08)";
   399→   399→      ctx.fillRect(0, scanY, W, 2);
   400→   400→
   401→   401→      if (playing && t < 1) {
   402→   402→        animRef.current = requestAnimationFrame(draw);
   403→   403→      } else if (playing && t >= 1) {
   404→   404→        startRef.current = 0;
   405→   405→        animRef.current = requestAnimationFrame(draw);
   406→   406→      }
   407→   407→    }
   408→   408→
   409→   409→    if (playing) {
   410→   410→      startRef.current = 0;
   411→   411→      animRef.current = requestAnimationFrame(draw);
   412→   412→    }
   413→   413→    return () => cancelAnimationFrame(animRef.current);
   414→   414→  }, [playing, styleIndex, imageUrl]);
   415→   415→
   416→   416→  return (
   417→   417→    <canvas
   418→   418→      ref={canvasRef}
   419→   419→      width={640}
   420→   420→      height={360}
   421→   421→      className="w-full h-full object-cover rounded-xl"
   422→   422→    />
   423→   423→  );
   424→   424→}

   426→   426→/* --- Audio Waveform Visualizer --- */
   427→   427→function AudioWaveform({ playing, color = "#a78bfa" }: { playing: boolean; color?: string }) {
   428→   428→  const canvasRef = useRef<HTMLCanvasElement>(null);
   429→   429→  const animRef = useRef<number>(0);
   430→   430→
   431→   431→  useEffect(() => {
   432→   432→    const canvas = canvasRef.current;
   433→   433→    if (!canvas) return;
   434→   434→    const ctx = canvas.getContext("2d")!;
   435→   435→    const W = canvas.width;
   436→   436→    const H = canvas.height;
   437→   437→    const BARS = 48;
   438→   438→    const barW = W / BARS;
   439→   439→    let phase = 0;
   440→   440→
   441→   441→    function draw() {
   442→   442→      ctx.clearRect(0, 0, W, H);
   443→   443→      phase += 0.04;
   444→   444→
   445→   445→      for (let i = 0; i < BARS; i++) {
   446→   446→        const x = i * barW;
   447→   447→        let h: number;
   448→   448→        if (playing) {
   449→   449→          const wave = Math.sin(phase + i * 0.3) * 0.4 + Math.sin(phase * 1.5 + i * 0.15) * 0.3 + Math.cos(phase * 0.7 + i * 0.5) * 0.3;
   450→   450→          h = Math.abs(wave) * H * 0.85 + H * 0.08;
   451→   451→        } else {
   452→   452→          h = H * 0.1;
   453→   453→        }
   454→   454→        const grad = ctx.createLinearGradient(x, H - h, x, H);
   455→   455→        grad.addColorStop(0, color);
   456→   456→        grad.addColorStop(1, color + "33");
   457→   457→        ctx.fillStyle = grad;
   458→   458→        ctx.fillRect(x + 1, H - h, barW - 2, h);
   459→   459→      }
   460→   460→      animRef.current = requestAnimationFrame(draw);
   461→   461→    }
   462→   462→    animRef.current = requestAnimationFrame(draw);
   463→   463→    return () => cancelAnimationFrame(animRef.current);
   464→   464→  }, [playing, color]);
   465→   465→
   466→   466→  return (
   467→   467→    <canvas ref={canvasRef} width={384} height={48} className="w-full h-12 rounded-lg" />
   468→   468→  );
   469→   469→}

   471→   471→/* --- Floating Particles --- */
   472→   472→function Particles({ count = 20 }: { count?: number }) {
   473→   473→  const canvasRef = useRef<HTMLCanvasElement>(null);
   474→   474→  const animRef = useRef<number>(0);
   475→   475→  const particlesRef = useRef<{ x: number; y: number; vx: number; vy: number; r: number; a: number }[]>([]);
   476→   476→
   477→   477→  useEffect(() => {
   478→   478→    const canvas = canvasRef.current;
   479→   479→    if (!canvas) return;
   480→   480→    const ctx = canvas.getContext("2d")!;
   481→   481→    const W = canvas.width;
   482→   482→    const H = canvas.height;
   483→   483→    particlesRef.current = Array.from({ length: count }, () => ({
   484→   484→      x: Math.random() * W, y: Math.random() * H,
   485→   485→      vx: (Math.random() - 0.5) * 0.5, vy: -Math.random() * 0.8 - 0.2,
   486→   486→      r: Math.random() * 2 + 0.5, a: Math.random() * 0.5 + 0.2,
   487→   487→    }));
   488→   488→
   489→   489→    function draw() {
   490→   490→      ctx.clearRect(0, 0, W, H);
   491→   491→      for (const p of particlesRef.current) {
   492→   492→        p.x += p.vx; p.y += p.vy;
   493→   493→        if (p.y < -5) { p.y = H + 5; p.x = Math.random() * W; }
   494→   494→        if (p.x < -5) p.x = W + 5;
   495→   495→        if (p.x > W + 5) p.x = -5;
   496→   496→        ctx.beginPath();
   497→   497→        ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
   498→   498→        ctx.fillStyle = `rgba(167,139,250,${p.a})`;
   499→   499→        ctx.fill();
   500→   500→      }
   501→   501→      animRef.current = requestAnimationFrame(draw);
   502→   502→    }
   503→   503→    animRef.current = requestAnimationFrame(draw);
   504→   504→    return () => cancelAnimationFrame(animRef.current);
   505→   505→  }, [count]);
   506→   506→
   507→   507→  return <canvas ref={canvasRef} width={200} height={200} className="absolute inset-0 w-full h-full pointer-events-none" />;
   508→   508→}

   511→   511→function AssembleProgress({ state, images }: { state: AssembleProgressState; images: Record<number, { url: string; final: boolean }> }) {
   512→   512→  const [elapsed, setElapsed] = useState(0);
   513→   513→  const [smoothPct, setSmoothPct] = useState(0);
   514→   514→
   515→   515→  useEffect(() => {
   516→   516→    const start = state.startTime || Date.now();
   517→   517→    const id = setInterval(() => setElapsed(Math.floor((Date.now() - start) / 1000)), 200);
   518→   518→    return () => clearInterval(id);
   519→   519→  }, [state.startTime]);
   520→   520→
   521→   521→  // Smooth progress animation
   522→   522→  useEffect(() => {
   523→   523→    const raw = state.total > 0
   524→   524→      ? Math.min(99, Math.round(((state.completed + (state.scenePct ?? 0) / 100) / state.total) * 100))
   525→   525→      : 5;
   526→   526→    const id = setInterval(() => {
   527→   527→      setSmoothPct(prev => {
   528→   528→        const diff = raw - prev;
   529→   529→        if (Math.abs(diff) < 0.5) return raw;
   530→   530→        return prev + diff * 0.08;
   531→   531→      });
   532→   532→    }, 30);

   558→   558→      </div>
   559→   559→
   560→   560→      <div className="relative p-5 sm:p-6 space-y-5">
   561→   561→        {/* Hero: Ken Burns preview + info */}
   562→   562→        <div className="grid md:grid-cols-[1fr_280px] gap-5">
