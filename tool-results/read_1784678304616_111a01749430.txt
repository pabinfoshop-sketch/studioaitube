   295→   295→function updateAssembleProgress(prev: AssembleProgressState, message: string, event?: AssembleEvent): AssembleProgressState {
   296→   296→  if (!event) return { ...prev, message };
   297→   297→  if (event.type === "log") {
   298→   298→    return { ...prev, logs: [...prev.logs, event.message].slice(-20), message };
   299→   299→  }
   300→   300→  if (event.type === "engine") {
   301→   301→    return { ...prev, message, phase: "Carregando engine", currentIndex: null, scenePct: 0 };
   302→   302→  }
   303→   303→  if (event.type === "scene-download") {
   304→   304→    return { ...prev, message, phase: "Baixando recursos", currentIndex: event.index, total: event.total, downloadItem: event.item, downloadPct: event.pct, scenePct: 0 };
   305→   305→  }
   306→   306→  if (event.type === "scene-start") {
   307→   307→    return { ...prev, message, phase: "Preparando cena", currentIndex: event.index, total: event.total, downloadItem: "", downloadPct: 0, scenePct: 10 };
   308→   308→  }
   309→   309→  if (event.type === "scene-render") {
   310→   310→    return { ...prev, message, phase: "Renderizando cena", currentIndex: event.index, total: event.total, downloadItem: "", downloadPct: 0, scenePct: 30, ffmpegPct: 0 };
   311→   311→  }
   312→   312→  if (event.type === "ffmpeg-progress") {
   313→   313→    const pct = Math.round((event.progress ?? 0) * 100);
   314→   314→    return { ...prev, message, phase: "Renderizando cena", currentIndex: event.index, total: prev.total, scenePct: 30 + pct * 0.7, ffmpegPct: pct };
   315→   315→  }
   316→   316→  if (event.type === "scene-done") {
   317→   317→    return { ...prev, message, phase: "Cena concluida", currentIndex: event.index, total: event.total, completed: Math.max(prev.completed, event.index + 1), downloadItem: "", downloadPct: 0, scenePct: 100 };
   318→   318→  }
   319→   319→  if (event.type === "concat") {
   320→   320→    return { ...prev, message, phase: "Unindo cenas", currentIndex: null, total: event.total, completed: event.total, scenePct: 0 };
   321→   321→  }
   322→   322→  if (event.type === "done") {
   323→   323→    return { ...prev, message, phase: "Video pronto", currentIndex: null, completed: prev.total, scenePct: 100 };
   324→   324→  }
   325→   325→  return { ...prev, message };
   326→   326→}

   328→   328→/* --- Ken Burns Canvas Preview --- */
   329→   329→function KenBurnsCanvas({ imageUrl, styleIndex, playing }: { imageUrl: string; styleIndex: number; playing: boolean }) {
   330→   330→  const canvasRef = useRef<HTMLCanvasElement>(null);
   331→   331→  const imgRef = useRef<HTMLImageElement | null>(null);
   332→   332→  const animRef = useRef<number>(0);
   333→   333→  const startRef = useRef(0);
   334→   334→
   335→   335→  useEffect(() => {
   336→   336→    const img = new Image();
   337→   337→    img.crossOrigin = "anonymous";
   338→   338→    img.src = imageUrl;
   339→   339→    img.onload = () => { imgRef.current = img; };
   340→   340→    imgRef.current = null;
   341→   341→  }, [imageUrl]);
   342→   342→
   343→   343→  useEffect(() => {
   344→   344→    const canvas = canvasRef.current;
   345→   345→    if (!canvas) return;
   346→   346→    const ctx = canvas.getContext("2d")!;
   347→   347→    const W = canvas.width;
   348→   348→    const H = canvas.height;
   349→   349→    const DURATION = 6000;
   350→   350→
   351→   351→    function draw(ts: number) {
   352→   352→      if (!startRef.current) startRef.current = ts;
   353→   353→      const elapsed = ts - startRef.current;
   354→   354→      const t = Math.min(elapsed / DURATION, 1);
   355→   355→      const img = imgRef.current;
   356→   356→
   357→   357→      ctx.fillStyle = "#000";
   358→   358→      ctx.fillRect(0, 0, W, H);
   359→   359→
   360→   360→      if (img) {
   361→   361→        const zoom = 1.0 + t * 0.3;
   362→   362→        const iw = W * zoom;
   363→   363→        const ih = H * zoom;
   364→   364→        let ox: number, oy: number;
   365→   365→        switch (styleIndex % 5) {
   366→   366→          case 0: ox = (W - iw) / 2; oy = (H - ih) / 2; break;
   367→   367→          case 1: ox = (W - iw) / 2; oy = (H - ih) / 2; break;
   368→   368→          case 2: ox = -(iw - W) * t; oy = (H - ih) / 2; break;
   369→   369→          case 3: ox = -(iw - W) * (1 - t); oy = (H - ih) / 2; break;
   370→   370→          default: ox = (W - iw) / 2; oy = -(ih - H) * t; break;
   371→   371→        }
   372→   372→        ctx.drawImage(img, ox, oy, iw, ih);
   373→   373→        // vignette
   374→   374→        const grad = ctx.createRadialGradient(W / 2, H / 2, W * 0.25, W / 2, H / 2, W * 0.7);
   375→   375→        grad.addColorStop(0, "rgba(0,0,0,0)");
   376→   376→        grad.addColorStop(1, "rgba(0,0,0,0.6)");
   377→   377→        ctx.fillStyle = grad;
   378→   378→        ctx.fillRect(0, 0, W, H);
   379→   379→      }
   380→   380→
   381→   381→      // cinematic bars
   382→   382→      ctx.fillStyle = "rgba(0,0,0,0.5)";
   383→   383→      ctx.fillRect(0, 0, W, H * 0.07);
   384→   384→      ctx.fillRect(0, H * 0.93, W, H * 0.07);
   385→   385→
   386→   386→      // scanline
   387→   387→      const scanY = (elapsed * 0.05) % H;
   388→   388→      ctx.fillStyle = "rgba(139,92,246,0.08)";
   389→   389→      ctx.fillRect(0, scanY, W, 2);
   390→   390→
   391→   391→      if (playing && t < 1) {
   392→   392→        animRef.current = requestAnimationFrame(draw);
   393→   393→      } else if (playing && t >= 1) {
   394→   394→        startRef.current = 0;
   395→   395→        animRef.current = requestAnimationFrame(draw);
   396→   396→      }
   397→   397→    }
   398→   398→
   399→   399→    if (playing) {
   400→   400→      startRef.current = 0;
   401→   401→      animRef.current = requestAnimationFrame(draw);
   402→   402→    }
   403→   403→    return () => cancelAnimationFrame(animRef.current);
   404→   404→  }, [playing, styleIndex, imageUrl]);
   405→   405→
   406→   406→  return (
   407→   407→    <canvas
   408→   408→      ref={canvasRef}
   409→   409→      width={640}
   410→   410→      height={360}
   411→   411→      className="w-full h-full object-cover rounded-xl"
   412→   412→    />
   413→   413→  );
   414→   414→}

   416→   416→/* --- Audio Waveform Visualizer --- */
   417→   417→function AudioWaveform({ playing, color = "#a78bfa" }: { playing: boolean; color?: string }) {
   418→   418→  const canvasRef = useRef<HTMLCanvasElement>(null);
   419→   419→  const animRef = useRef<number>(0);
   420→   420→
   421→   421→  useEffect(() => {
   422→   422→    const canvas = canvasRef.current;
   423→   423→    if (!canvas) return;
   424→   424→    const ctx = canvas.getContext("2d")!;
   425→   425→    const W = canvas.width;
   426→   426→    const H = canvas.height;
   427→   427→    const BARS = 48;
   428→   428→    const barW = W / BARS;
   429→   429→    let phase = 0;
   430→   430→
   431→   431→    function draw() {
   432→   432→      ctx.clearRect(0, 0, W, H);
   433→   433→      phase += 0.04;
   434→   434→
   435→   435→      for (let i = 0; i < BARS; i++) {
   436→   436→        const x = i * barW;
   437→   437→        let h: number;
   438→   438→        if (playing) {
   439→   439→          const wave = Math.sin(phase + i * 0.3) * 0.4 + Math.sin(phase * 1.5 + i * 0.15) * 0.3 + Math.cos(phase * 0.7 + i * 0.5) * 0.3;
   440→   440→          h = Math.abs(wave) * H * 0.85 + H * 0.08;
   441→   441→        } else {
   442→   442→          h = H * 0.1;
   443→   443→        }
   444→   444→        const grad = ctx.createLinearGradient(x, H - h, x, H);
   445→   445→        grad.addColorStop(0, color);
   446→   446→        grad.addColorStop(1, color + "33");
   447→   447→        ctx.fillStyle = grad;
   448→   448→        ctx.fillRect(x + 1, H - h, barW - 2, h);
   449→   449→      }
   450→   450→      animRef.current = requestAnimationFrame(draw);
   451→   451→    }
   452→   452→    animRef.current = requestAnimationFrame(draw);
   453→   453→    return () => cancelAnimationFrame(animRef.current);
   454→   454→  }, [playing, color]);
   455→   455→
   456→   456→  return (
   457→   457→    <canvas ref={canvasRef} width={384} height={48} className="w-full h-12 rounded-lg" />
   458→   458→  );
   459→   459→}

   461→   461→/* --- Floating Particles --- */
   462→   462→function Particles({ count = 20 }: { count?: number }) {
   463→   463→  const canvasRef = useRef<HTMLCanvasElement>(null);
   464→   464→  const animRef = useRef<number>(0);
   465→   465→  const particlesRef = useRef<{ x: number; y: number; vx: number; vy: number; r: number; a: number }[]>([]);
   466→   466→
   467→   467→  useEffect(() => {
   468→   468→    const canvas = canvasRef.current;
   469→   469→    if (!canvas) return;
   470→   470→    const ctx = canvas.getContext("2d")!;
   471→   471→    const W = canvas.width;
   472→   472→    const H = canvas.height;
   473→   473→    particlesRef.current = Array.from({ length: count }, () => ({
   474→   474→      x: Math.random() * W, y: Math.random() * H,
   475→   475→      vx: (Math.random() - 0.5) * 0.5, vy: -Math.random() * 0.8 - 0.2,
   476→   476→      r: Math.random() * 2 + 0.5, a: Math.random() * 0.5 + 0.2,
   477→   477→    }));
   478→   478→
   479→   479→    function draw() {
   480→   480→      ctx.clearRect(0, 0, W, H);
   481→   481→      for (const p of particlesRef.current) {
   482→   482→        p.x += p.vx; p.y += p.vy;
   483→   483→        if (p.y < -5) { p.y = H + 5; p.x = Math.random() * W; }
   484→   484→        if (p.x < -5) p.x = W + 5;
   485→   485→        if (p.x > W + 5) p.x = -5;
   486→   486→        ctx.beginPath();
   487→   487→        ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
   488→   488→        ctx.fillStyle = `rgba(167,139,250,${p.a})`;
   489→   489→        ctx.fill();
   490→   490→      }
   491→   491→      animRef.current = requestAnimationFrame(draw);
   492→   492→    }
   493→   493→    animRef.current = requestAnimationFrame(draw);
   494→   494→    return () => cancelAnimationFrame(animRef.current);
   495→   495→  }, [count]);
   496→   496→
   497→   497→  return <canvas ref={canvasRef} width={200} height={200} className="absolute inset-0 w-full h-full pointer-events-none" />;
   498→   498→}
